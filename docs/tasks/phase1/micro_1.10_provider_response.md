```go
package provider

import "time"

// =============================================================================
// Response — Full (non-streaming) response from provider
// =============================================================================

// Response represents a complete response from an AI provider.
//
// It is returned by Provider.Send() after the AI finishes generating.
type Response struct {
	// ID is a unique identifier for this response.
	// Generated by the provider (e.g., "chatcmpl-abc123").
	ID string `json:"id"`

	// Content is the text content of the response.
	// May be empty if the AI only returned tool calls.
	Content string `json:"content"`

	// ToolCalls contains tool/function calls made by the AI.
	// Empty if the AI did not call any tools.
	ToolCalls []ToolCall `json:"tool_calls,omitempty"`

	// FinishReason indicates why the AI stopped generating.
	//
	// Common values across providers:
	//   "stop"       — Normal completion (AI finished its response)
	//   "tool_calls" — AI wants to call tools (check ToolCalls field)
	//   "max_tokens" — Response was cut off due to token limit
	//   "error"      — Provider encountered an error
	//
	// WHY string instead of enum?
	// → Each provider returns different values. Forcing an enum would
	//   require updating it every time a new provider is added.
	//   String is flexible and forward-compatible.
	FinishReason string `json:"finish_reason"`

	// Usage tracks token consumption for this request.
	Usage Usage `json:"usage"`

	// Model is the actual model that processed the request.
	// May differ from the requested model (e.g., "gemini-2.5-pro-preview"
	// when "gemini-2.5-pro" was requested).
	Model string `json:"model"`

	// CreatedAt is when the response was generated.
	CreatedAt time.Time `json:"created_at"`
}

// =============================================================================
// Usage — Token consumption tracking
// =============================================================================

// Usage tracks token consumption for cost monitoring and optimization.
//
// Why track usage?
// 1. Cost monitoring: Each token costs money (Gemini, Claude, OpenAI all charge per token)
// 2. Budget enforcement: Stop missions when budget is exceeded
// 3. Optimization: Identify agents/tasks that consume too many tokens
// 4. Analytics: Track token usage trends over time
type Usage struct {
	// PromptTokens is the number of tokens in the request (input).
	PromptTokens int `json:"prompt_tokens"`

	// CompletionTokens is the number of tokens in the response (output).
	CompletionTokens int `json:"completion_tokens"`

	// TotalTokens is PromptTokens + CompletionTokens.
	// Some providers only return TotalTokens without the breakdown.
	TotalTokens int `json:"total_tokens"`
}

// Add merges another Usage into this one (for aggregation).
//
// Example:
//
//	totalUsage := Usage{}
//	totalUsage.Add(task1Usage)
//	totalUsage.Add(task2Usage)
func (u *Usage) Add(other Usage) {
	u.PromptTokens += other.PromptTokens
	u.CompletionTokens += other.CompletionTokens
	u.TotalTokens += other.TotalTokens
}

// IsZero returns true if no tokens were used.
func (u *Usage) IsZero() bool {
	return u.TotalTokens == 0
}

// =============================================================================
// StreamChunk — Single chunk in a streaming response
// =============================================================================

// StreamChunk represents a single piece of a streaming response.
//
// Provider.Stream() returns a channel of StreamChunks.
// The consumer should read chunks until Done is true or the channel is closed.
//
// Example:
//
//	stream, err := provider.Stream(ctx, req)
//	if err != nil { ... }
//
//	var fullContent strings.Builder
//	for chunk := range stream {
//	    if chunk.Error != nil {
//	        // Handle error mid-stream
//	        break
//	    }
//	    fullContent.WriteString(chunk.Delta)
//	    if chunk.Done {
//	        break
//	    }
//	}
type StreamChunk struct {
	// Delta is the new text content appended in this chunk.
	// Concatenating all Deltas gives the full response content.
	Delta string `json:"delta"`

	// ToolCalls contains partial tool call information.
	// Tool call data may span multiple chunks.
	// Consumer should accumulate until Done is true.
	ToolCalls []ToolCall `json:"tool_calls,omitempty"`

	// Done indicates this is the last chunk in the stream.
	// After receiving a chunk with Done=true, no more chunks will follow.
	//
	// WHY is this important?
	// → Consumer goroutine needs to know when to stop reading.
	// → Without this flag, consumer waits forever → goroutine leak.
	Done bool `json:"done"`

	// FinishReason is populated in the last chunk (when Done=true).
	// Same values as Response.FinishReason.
	FinishReason string `json:"finish_reason,omitempty"`

	// Usage is populated in the last chunk (when Done=true).
	// Some providers only provide usage information at the end.
	Usage *Usage `json:"usage,omitempty"`

	// Error is populated if an error occurred during streaming.
	// When Error is non-nil, the stream should be considered terminated.
	// This is for errors that occur AFTER the stream has started
	// (e.g., network disconnect mid-stream).
	Error error `json:"-"` // Not serialized — error interface can't be JSON marshalled
}

// =============================================================================
// Response Helper Methods
// =============================================================================

// HasToolCalls returns true if the response contains tool calls.
func (r *Response) HasToolCalls() bool {
	return len(r.ToolCalls) > 0
}

// IsComplete returns true if the AI finished naturally (not cut off).
func (r *Response) IsComplete() bool {
	return r.FinishReason == "stop" || r.FinishReason == "end_turn"
}

// IsTruncated returns true if the response was cut off due to token limit.
func (r *Response) IsTruncated() bool {
	return r.FinishReason == "max_tokens" || r.FinishReason == "length"
}

// WantsToolCall returns true if the AI stopped because it wants to call tools.
func (r *Response) WantsToolCall() bool {
	return r.FinishReason == "tool_calls" || r.FinishReason == "tool_use"
}

// ToMessage converts the response into a Message for conversation history.
// This is useful for building multi-turn conversations.
func (r *Response) ToMessage() Message {
	return Message{
		Role:      RoleAssistant,
		Content:   r.Content,
		ToolCalls: r.ToolCalls,
	}
}
```

## Quy tắc
1. `FinishReason` là `string` — mỗi provider dùng giá trị khác nhau
2. `Usage` có `Add()` method — aggregate token usage across multiple tasks
3. `StreamChunk.Error` có tag `json:"-"` — `error` interface KHÔNG serialize được
4. `StreamChunk.Done` PHẢI có — consumer cần biết khi nào stream kết thúc
5. `StreamChunk.Usage` là `*Usage` (pointer) — chỉ populated ở chunk cuối cùng
6. Helper methods handle NHIỀU giá trị FinishReason (OpenAI: "stop", Claude: "end_turn")

## ⚠️ Pitfalls cần tránh
1. **`error` interface và JSON**: `json.Marshal(err)` trả về `{}` vì error interface chỉ có `Error()` method, không có exported fields. Dùng `json:"-"` tag và truyền error qua channel, KHÔNG qua JSON.
2. **StreamChunk accumulation**: Tool calls có thể split across multiple chunks. Consumer phải accumulate, KHÔNG xử lý partial tool call.
3. **Stream termination**: 3 cách stream kết thúc:
   - `Done = true` → normal
   - `Error != nil` → error mid-stream
   - Channel closed → producer died (check with `for chunk, ok := range ch`)
4. **Response.CreatedAt**: Dùng `time.Time` thay vì `int64` timestamp. JSON serialization: `time.Time` marshal thành `"2006-01-02T15:04:05Z"` (RFC 3339).
5. **Usage.TotalTokens**: Một số providers chỉ trả `TotalTokens` mà không breakdown. Check `PromptTokens == 0 && CompletionTokens == 0 && TotalTokens > 0`.
