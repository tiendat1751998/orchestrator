```go
// Package provider defines the contract for AI provider integrations.
// All providers (Antigravity, Gemini API, Claude, Ollama, etc.)
// must implement the Provider interface defined in this package.
package provider

import "encoding/json"

// =============================================================================
// Message Roles
// =============================================================================

// Role represents the role of a message sender in a conversation.
type Role string

const (
	// RoleSystem is the system prompt that defines AI behavior.
	// Example: "You are a backend developer. Write clean Go code."
	RoleSystem Role = "system"

	// RoleUser is a message from the human user or the orchestrator.
	RoleUser Role = "user"

	// RoleAssistant is a message from the AI model.
	RoleAssistant Role = "assistant"

	// RoleTool is a message containing tool execution results.
	// This is sent back to the AI after executing a tool call.
	RoleTool Role = "tool"
)

// IsValid checks if the role is one of the defined constants.
func (r Role) IsValid() bool {
	switch r {
	case RoleSystem, RoleUser, RoleAssistant, RoleTool:
		return true
	default:
		return false
	}
}

// String returns the string representation.
func (r Role) String() string {
	return string(r)
}

// =============================================================================
// Message
// =============================================================================

// Message represents a single message in a conversation.
//
// Conversation flow example:
//
//	messages := []Message{
//	    {Role: RoleSystem, Content: "You are a Go developer."},
//	    {Role: RoleUser, Content: "Write a hello world program."},
//	    {Role: RoleAssistant, Content: "```go\npackage main...```", ToolCalls: []ToolCall{...}},
//	    {Role: RoleTool, Content: `{"output": "file created"}`, ToolCallID: "call_123"},
//	}
type Message struct {
	// Role identifies who sent this message.
	Role Role `json:"role"`

	// Content is the text content of the message.
	// For RoleAssistant with tool calls, this may be empty.
	Content string `json:"content"`

	// ToolCalls is populated when the AI wants to call tools.
	// Only present in RoleAssistant messages.
	// Can contain multiple tool calls (parallel tool calling).
	ToolCalls []ToolCall `json:"tool_calls,omitempty"`

	// ToolCallID links a RoleTool message to the original ToolCall.
	// Only present in RoleTool messages.
	// Must match the ID of a ToolCall in a previous RoleAssistant message.
	ToolCallID string `json:"tool_call_id,omitempty"`

	// Name is an optional identifier for the message sender.
	// Useful for multi-agent conversations to identify which agent sent what.
	Name string `json:"name,omitempty"`
}

// =============================================================================
// Tool Call
// =============================================================================

// ToolCall represents the AI's request to execute a tool/function.
//
// Example:
//
//	ToolCall{
//	    ID:   "call_abc123",
//	    Name: "read_file",
//	    Args: json.RawMessage(`{"path": "/src/main.go", "start_line": 1, "end_line": 50}`),
//	}
type ToolCall struct {
	// ID is a unique identifier for this tool call.
	// Used to match the tool result back to this call.
	// Generated by the provider/AI.
	ID string `json:"id"`

	// Name is the name of the tool to execute.
	// Must match a registered tool name (e.g., "read_file", "git_commit").
	Name string `json:"name"`

	// Args contains the arguments for the tool as raw JSON.
	//
	// WHY json.RawMessage instead of map[string]any?
	// 1. Preserves JSON number precision (map[string]any converts all numbers to float64)
	// 2. Each tool can unmarshal into its own typed struct
	// 3. Avoids intermediate map allocation
	//
	// Example usage in tool implementation:
	//   type ReadFileArgs struct {
	//       Path      string `json:"path"`
	//       StartLine int    `json:"start_line"`
	//   }
	//   var args ReadFileArgs
	//   json.Unmarshal(toolCall.Args, &args)
	Args json.RawMessage `json:"arguments"`
}

// =============================================================================
// Helper Functions
// =============================================================================

// NewSystemMessage creates a system message.
func NewSystemMessage(content string) Message {
	return Message{Role: RoleSystem, Content: content}
}

// NewUserMessage creates a user message.
func NewUserMessage(content string) Message {
	return Message{Role: RoleUser, Content: content}
}

// NewAssistantMessage creates an assistant message with text content.
func NewAssistantMessage(content string) Message {
	return Message{Role: RoleAssistant, Content: content}
}

// NewToolResultMessage creates a tool result message.
//
// Parameters:
//   - toolCallID: must match the ID of the original ToolCall
//   - content: the output of the tool execution (as string)
func NewToolResultMessage(toolCallID string, content string) Message {
	return Message{
		Role:       RoleTool,
		Content:    content,
		ToolCallID: toolCallID,
	}
}

// HasToolCalls returns true if this is an assistant message with tool calls.
func (m *Message) HasToolCalls() bool {
	return m.Role == RoleAssistant && len(m.ToolCalls) > 0
}
```

## Quy tắc
1. `json.RawMessage` cho ToolCall Args — xem comment chi tiết trong code
2. JSON tag `"arguments"` (không phải `"args"`) — đây là format chuẩn OpenAI/Gemini
3. Helper functions (`NewSystemMessage`, etc.) giảm boilerplate
4. `HasToolCalls()` pointer receiver — tránh copy struct lớn

## ⚠️ Pitfalls cần tránh
1. **ToolCall.Args JSON field name**: OpenAI dùng `"arguments"`, Gemini dùng `"args"`. CHỌN `"arguments"` vì phổ biến hơn. Provider adapter sẽ mapping nếu cần.
2. **Multiple ToolCalls**: AI có thể gọi NHIỀU tools cùng lúc (parallel tool calling). `ToolCalls` là slice, KHÔNG phải single value.
3. **ToolCallID matching**: Mỗi tool result PHẢI có `ToolCallID` khớp với `ToolCall.ID`. Nếu không khớp → AI sẽ confused → output sai.
4. **Empty Content**: Khi AI gọi tools, `Content` có thể TRỐNG. KHÔNG check `Content == ""` để validate message — check `Role` thay.
5. **Name field**: Optional, dùng cho multi-agent conversations. KHÔNG bắt buộc cho single-agent.
